# Mise en forme de la date
#### ¬®pit avoir toutes les ann√©es tout format de quatre charact√®res. Ex: ( 5 -> 005 -> 05 -> 2005)  , (16 -> 0016, 16 -> 2016)
df_acc_usr['an01']  = ('00' + df_acc_usr['an'].map(str) )
df_acc_usr['an02']  = ('20' + df_acc_usr['an01'].str[-2:])

#### Pour avoir tout les mois sous format de deux charact√®res. Ex: ( 1 -> 001 -> 01) , (13 -> 0013 -> 13)
df_acc_usr['mois01']  = ('00' + df_acc_usr['mois'].map(str) ) 
df_acc_usr['mois02']  = ( df_acc_usr['mois01'].str[-2:] )

## Pour avoir tout les jours sous format de deux charact√®res. Ex: ( 3 -> 003 -> 03) , (15 -> 0015 -> 15)
df_acc_usr['jour01']  = ('00' + df_acc_usr['jour'].map(str) )
df_acc_usr['jour02']  = ( df_acc_usr['jour01'].str[-2:] )

## Pour avoir tout les jours sous format de quatre charact√®res. Ex: ( 3 -> 0003 -> 0003) , (1250 -> 0001250 -> 1250) , (230 -> 000230 -> 0230)
df_acc_usr['hrmn01']  = ('000' + df_acc_usr['hrmn'].map(str) )

# Choix des colonnes √† utiliser pour la pr√©diction
feature_cols = list(df_acc_usr.columns) # Lister toutes les colonnes
feature_cols.remove('Num_Acc') # Colonne non utilis√©e
feature_cols.remove('Date') # Colonne non utilis√©e
feature_cols.remove('grav') # Colonne non utilis√©e
feature_cols.remove('env1')  # Colonne non utilis√©e

# DF des observations
X = imputed_df_acc_usr[feature_cols] 

# DF cible
y = df_acc_usr['grav'] 

# D√©couper le dataset en aprentissage et teste
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=1) # 70% training and 30% test

# Cr√©ation d'un objet arbre de d√©cision, crit√®re de division est l'entropy
clf = DecisionTreeClassifier(max_depth=6, criterion = "entropy")
# Aprentissage d'arbre de d√©cision
clf = clf.fit(X_train,y_train)

# Pr√©diction sur le dataset
y_pred = clf.predict(X_test)

# Calcul de l'accuracy de notre mod√®le
print("Accuracy:",metrics.accuracy_score(y_test, y_pred))
